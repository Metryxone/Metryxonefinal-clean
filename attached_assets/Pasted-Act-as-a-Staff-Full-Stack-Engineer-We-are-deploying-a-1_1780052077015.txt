export function rephraseForProxy(spec: StructuralClarityQuestion, subjectName: string): StructuralClarityQuestion {
  const cloned: StructuralClarityQuestion = JSON.parse(JSON.stringify(spec));
  let text = cloned.question_text;

  // 1. Structural substitution matching common English clauses cleanly
  if (text.toLowerCase().includes("how confident are you")) {
    text = text.replace(/are you/i, `is ${subjectName}`);
  } else if (text.toLowerCase().includes("do you feel")) {
    text = text.replace(/do you feel/i, `does ${subjectName} feel`);
  } else if (text.toLowerCase().includes("are you able to")) {
    text = text.replace(/are you able to/i, `is ${subjectName} able to`);
  } else {
    // Structural Fallback matching standard pronouns
    text = text.replace(/\byou are\b/i, `${subjectName} is`);

  // 2. Trailing Pronoun and Possessive Sweeps

  text = text.replace(/\bis is\b/gi, "is");
