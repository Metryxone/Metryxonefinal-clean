import React, { useState, useEffect } from 'react';

export interface StructuralClarityQuestion {
  id: string;
  evaluation_type: 'LIKERT' | 'RANKING';
  question_text: string;
  options: {
    key: string;
    option_text: string;
    score_weight: number;
  }[];
}

interface CapadexClarifyPhaseProps {
  currentQuestion: StructuralClarityQuestion;
  onSaveSelection: (questionId: string, selections: string[]) => void;
  submitting?: boolean;
}

export const CapadexClarifyPhase: React.FC<CapadexClarifyPhaseProps> = ({
  currentQuestion,
  onSaveSelection,
  submitting = false
}) => {
  const [selectedKeys, setSelectedKeys] = useState<string[]>([]);

  // Clear component state automatically when moving to a new question item
  useEffect(() => {
    setSelectedKeys([]);
  }, [currentQuestion.id]);

  const handleSelectionClick = (clickedKey: string) => {
    if (submitting) return;

    if (currentQuestion.evaluation_type === 'LIKERT') {
      // Likert Rule: Immediate single-action commit lifecycle
      setSelectedKeys([clickedKey]);
      onSaveSelection(currentQuestion.id, [clickedKey]);
    } else if (currentQuestion.evaluation_type === 'RANKING') {
      // Ranking Rule: Interactive selection index toggling stack
      let updated = [...selectedKeys];
      if (updated.includes(clickedKey)) {
        updated = updated.filter(k => k !== clickedKey);
      } else {
        updated.push(clickedKey);
      }
      setSelectedKeys(updated);
    }
  };

  return (
    <div className="clarify-phase-container max-w-2xl mx-auto p-6 bg-white rounded-xl shadow-sm border border-gray-100" role="region" aria-label="Clarity Assessment Step">
      <h2 className="text-xl font-semibold text-gray-900 mb-2">{currentQuestion.question_text}</h2>
      
      
      <p className="text-sm text-gray-500 mb-6">
        {currentQuestion.evaluation_type === 'LIKERT'
          ? "Please select the option that fits best."
          : "Tap the options below in your relative order of importance (1 is most important)."}
      </p>

      <div className="options-stack space-y-3">
        {currentQuestion.options.map((opt) => {
          const rankIndex = selectedKeys.indexOf(opt.key);
          const isSelected = rankIndex > -1;

          return (
            <button
              key={opt.key}
              disabled={submitting}
              onClick={() => handleSelectionClick(opt.key)}
              className={`w-full p-4 text-left border rounded-lg transition-all flex justify-between items-center ${
                isSelected 
                  ? 'border-indigo-600 bg-indigo-50 text-indigo-900 font-medium' 
                  : 'border-gray-200 bg-white text-gray-700 hover:border-gray-300'
              } disabled:opacity-60`}
            >
              <span>{opt.option_text}</span>
              
              
              {isSelected && currentQuestion.evaluation_type === 'RANKING' && (
                <span className="bg-indigo-600 text-white w-6 h-6 rounded-full flex items-center justify-center text-xs font-bold shadow-sm">
                  {rankIndex + 1}
                </span>
              )}
              {isSelected && currentQuestion.evaluation_type === 'LIKERT' && (
                <span className="w-3 h-3 bg-indigo-600 rounded-full shadow-sm" />
              )}
            </button>
          );
        })}
      </div>

      
      {currentQuestion.evaluation_type === 'RANKING' && (
        <button
          disabled={submitting || selectedKeys.length === 0}
          onClick={() => onSaveSelection(currentQuestion.id, selectedKeys)}
          className="mt-6 w-full py-3 bg-gray-900 hover:bg-gray-800 text-white font-medium rounded-lg transition-colors disabled:opacity-40 disabled:cursor-not-allowed shadow-sm"
        >
          {submitting ? 'Saving selections...' : 'Confirm Selections'}
        </button>
      )}
    </div>
  );
};
