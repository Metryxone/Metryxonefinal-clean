## ⚙️ main.py (Core API)

```python
from fastapi import FastAPI
from pydantic import BaseModel
from rag import get_rag_response
from memory import save_chat, get_history

app = FastAPI()

class ChatRequest(BaseModel):
    user_id: str
    message: str

@app.post("/chat")
def chat(req: ChatRequest):
    history = get_history(req.user_id)
    
    response = get_rag_response(req.message, history)
    
    save_chat(req.user_id, req.message, response)
    
    return {"response": response}
```

## 🧠 rag.py (Vector DB + LLM)

```python
import faiss
import numpy as np
from openai import OpenAI

client = OpenAI()

# Dummy embedding function (replace with OpenAI embeddings)
def embed(text):
    return np.random.rand(768).astype("float32")

# Initialize FAISS
index = faiss.IndexFlatL2(768)

documents = [
    "CPR steps include chest compressions and rescue breaths",
    "Employability skills include communication and problem solving",
    "High glucose after meal is due to carbohydrate load"
]

doc_embeddings = np.array([embed(doc) for doc in documents])
index.add(doc_embeddings)

def retrieve(query):
    q_emb = np.array([embed(query)])
    D, I = index.search(q_emb, k=2)
    return [documents[i] for i in I[0]]

def get_rag_response(query, history):
    context = retrieve(query)
    
    prompt = f"""
You are MetryxOne AI Assistant.

Context:
{context}

Chat History:
{history}

User:
{query}

Respond like a human expert.
"""

    response = client.chat.completions.create(
        model="gpt-4.1-mini",
        messages=[{"role": "user", "content": prompt}]
    )

    return response.choices[0].message.content
```

## 💾 memory.py (Chat Memory)

```python
from db import conn, cursor

def save_chat(user_id, message, response):
    cursor.execute(
        "INSERT INTO chats VALUES (?, ?, ?)",
        (user_id, message, response)
    )
    conn.commit()

def get_history(user_id):
    cursor.execute(
        "SELECT message, response FROM chats WHERE user_id=? ORDER BY rowid DESC LIMIT 5",
        (user_id,)
    )
    rows = cursor.fetchall()
    
    history = ""
    for m, r in rows:
        history += f"User: {m}\nAssistant: {r}\n"
    
    return history
```

## app.jsx

```javascript
import React, { useState } from "react";
import { createRoot } from "react-dom/client";

function App() {
  const [message, setMessage] = useState("");
  const [chat, setChat] = useState([]);

  const sendMessage = async () => {
    const res = await fetch("/chat", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        user_id: "user123",
        message
      })
    });

    const data = await res.json();

    setChat([...chat, { user: message, bot: data.response }]);
    setMessage("");
  };

  return (
    <div style={{ padding: 20 }}>
      <h2>MetryxOne AI Assistant</h2>

      <div>
        {chat.map((c, i) => (
          <div key={i}>
            <p><b>You:</b> {c.user}</p>
            <p><b>AI:</b> {c.bot}</p>
          </div>
        ))}
      </div>

      <input
        value={message}
        onChange={(e) => setMessage(e.target.value)}
      />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}

createRoot(document.getElementById("root")).render(<App />);
```
